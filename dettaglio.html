    function aggiornaTermostato(temp, isEmpty) {
    const minT = 16; const maxT = 30;
    let val = temp; if(val < minT) val = minT; if(val > maxT) val = maxT;
    const progress = (val - minT) / (maxT - minT);
    const dashOffset = isEmpty ? 660 : 660 - (progress * 660);
    document.getElementById('pro-arc-path').style.strokeDashoffset = dashOffset;
    const angle = -135 + (progress * 270);
    document.getElementById('pro-arc-thumb').style.transform = `rotate(${angle}deg)`;
    }

    });
    }
